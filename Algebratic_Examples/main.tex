   \textbf{Example: val x = 1; return x;}
    	\begin{center}
		\begin{prooftree}
			\AxiomC{}
			\RightLabel{[LIT]}
			\UnaryInfC{$\Gamma \vdash 1 : \tau _0 $}
			\RightLabel{[RET]}
			\UnaryInfC{$\Gamma \vdash $ return 1 : $\tau_0 \:\:|\:\:  \emptyset $}
			
			\AxiomC{x : $\tau_1 \in \Gamma$ , x : $\tau_0$}
			\RightLabel{[VAR]}
			\UnaryInfC{$\Gamma \vdash x :\tau_0 \vdash x: \tau_1$}
			\RightLabel{[RET]}
			\UnaryInfC{$\Gamma, x:\tau_0 \vdash $ return x : $\tau_1 \:\:|\:\:  C_1 $}
		
			\RightLabel{[VAL]}
			\BinaryInfC{$\Gamma \vdash$ val x = return 1; return x : $\tau_1 \:\:|\:\: \emptyset \cup C_1$}
		\end{prooftree}
	\end{center}
	$\newline$
	Replace $\tau_0$ and $\tau_1$ with Int, $ C_1$ with $\emptyset$

  \textbf{Example: def Identity = \{(x : Int, $\emptyset$) $\Rightarrow$ return x\}; Identity(1,$\emptyset$);}
  
    	\begin{center}
		\begin{prooftree}
			\AxiomC{x : $\tau_1 \in \Gamma , (x:Int, \emptyset)$}
			\RightLabel{[VAR]}
			\UnaryInfC{$\Gamma, (x:Int, \emptyset) \vdash x: \tau_1 |\:  C $}
			\RightLabel{[RET]}
			\UnaryInfC{$\Gamma ,(x:Int, \emptyset)\vdash $ return x : $\tau_1 \:\:|\:\:  C \cup \overrightarrow {g_j}$}
			\RightLabel{[BLOCK]}
			\UnaryInfC{$\Gamma \vdash$ \{(x : $Int$,$\emptyset) \Rightarrow $return x\} : $\sigma \:|\: C $}
			
			
			\AxiomC{$Identity:^C(\tau'',\emptyset)\rightarrow \tau \in \Gamma, Identity:^{C'}$}
			\RightLabel{[LIT]}
			\UnaryInfC{$\Gamma, Identity:^{C'} \vdash Identity\::\:(\tau'', \emptyset) \rightarrow \tau \:|\: C$}
			
			\AxiomC{}
			\RightLabel{[LIT]}
			\UnaryInfC{$\Gamma , Identity:^{C'}\vdash 1 : \tau \:|\: C$}
			
			\AxiomC{$\Gamma, Identity:^{C'}\vdash \emptyset : \sigma | C$}
			
			\RightLabel{[APP]}
			\TrinaryInfC{$\Gamma, Identity:^{C'} \sigma \vdash Identity(1, \emptyset) : \tau \:|\: C $}
		
			\RightLabel{[DEF]}
			\BinaryInfC{$\Gamma \vdash$ def Identity = \{(x : Int, $\emptyset$) $\Rightarrow$ return x\}; Identity(1,$\emptyset$)}
		\end{prooftree}
	\end{center}
